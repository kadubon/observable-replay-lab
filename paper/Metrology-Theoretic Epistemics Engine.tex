\begin{abstract}\noindent
\STE{}~\cite{STE2026} provides auditable growth laws under \emph{no-meta} and \emph{observable-only} constraints.
This supplement formalizes the complementary bottleneck: \emph{identifiability supply}.
\MTE{} defines observability capital, observability regimes, and a fail-closed observability credit $\deltao(t)$ driven by replay-verifiable ECR proxies.

This paper addresses extreme-horizon failure modes:
(i) deductive exhaustion (G\"odel-type unprovability or budget-limited proof search) via an empirical reflection gate with explicit soundness-risk accounting;
(ii) ontological phase transitions via rupture detection and meta-morphism escrow;
(iii) cosmological depletion of independent witness supply via a Page-limit discount;
(iv) trustless extortion via cryptographic myopia bindings and mixed-strategy leverage bounds; and
(v) retroactive cryptographic decay via proactive proof freshening with proof-carrying data wrappers.

Relative to \STE{}, this paper enforces a strict non-overlap contract: strict-credit semantics, reserve/finality semantics, and reflective trust boundaries remain authoritative in \STE{}, while \MTE{} contributes computable observability operators, compositional risk accounting, bridge-holonomy distortion control, and a coupled credit supermartingale law.
\end{abstract}

\section{Formal Setting: Uncertainty, Regularity, and Computability}\label{sec:formal}
Let $\Et$ denote the set of explanations consistent with strict artifacts up to time $t$.

\begin{definition}[Observable probability space]\DefLabel{def:observable_probability_space}\label{def:observable_probability_space}
All stochastic claims in \MTE{} are interpreted on a pinned probability space $(\Omega,\mathcal{F},\Prob)$ with public filtration $\Filtration$ generated from strict artifacts.
A random object is admissible in certified claims only if its measurability/adaptedness status relative to $\Filtration$ is declared and replay-verifiable.
\end{definition}

\begin{definition}[Audit-stochastic admissibility]\DefLabel{def:audit_stochastic_admissibility}\label{def:audit_stochastic_admissibility}
An \MTE{} stochastic certificate is admissible only if it publishes:
(i) integrability bounds for all used conditional expectations,
(ii) stopping-time validity conditions for any adaptive rule,
and (iii) the verifier identifier for the martingale/e-process calibration.
Missing items fail closed.
\end{definition}

\begin{definition}[Canonical window artifact reducer]\DefLabel{def:canonical_window_artifact_reducer}\label{def:canonical_window_artifact_reducer}
Let $\mathcal{L}$ be the append-only strict event log.
For window $t$, define the canonical bundle $\mathcal{A}_t:=\mathsf{Reduce}(\mathcal{L},t)$ by:
\begin{enumerate}
\item filter events with \texttt{window\_id}=t and valid signature/hash/type checks;
\item normalize all typed fields via pinned parser profile and unit profile;
\item sort accepted events lexicographically by $(\texttt{timestamp},\texttt{event\_id},\texttt{canonical\_hash})$;
\item for singleton event families, keep the unique maximal element under this order;
\item for multiset event families, keep all elements as a hash-sorted list.
\end{enumerate}
If any required field is missing or typed normalization fails, $\mathsf{Reduce}$ returns \texttt{FAIL\_CLOSED}.
\end{definition}

\begin{theorem}[Canonical bundle determinism]\DefLabel{thm:canonical_bundle_determinism}\label{thm:canonical_bundle_determinism}
For fixed parser/unit/signature profiles, Definition~\ref{def:canonical_window_artifact_reducer} yields a unique $\mathcal{A}_t$ for each window $t$.
Hence all downstream quantities that are deterministic functions of $\mathcal{A}_t$ are replay-invariant.
\end{theorem}

\begin{definition}[Dual risk ledger]\DefLabel{def:dual_risk_ledger}\label{def:dual_risk_ledger}
For each window $t$, define channel debits
\[
\alpha_{\mathrm{ref}}(t),\ \alpha_{\mathrm{avail}}(t),\ \alpha_{\mathrm{crypto}}(t),\ \alpha_{\mathrm{deni}}(t)\in[0,1]
\]
and define
\[
\alpha_{\Sigma}(t):=\alpha_{\mathrm{ref}}(t)+\alpha_{\mathrm{avail}}(t)+\alpha_{\mathrm{crypto}}(t)+\alpha_{\mathrm{deni}}(t).
\]
The window is admissible only if $\alpha_{\Sigma}(t)\le 1$.
The reflection ledger debit $\erisk(t)$ is tied by a pinned map $\Phi_t$:
\[
\erisk(t)=\Phi_t\!\big(\alpha_{\mathrm{ref}}(t),\alpha_{\mathrm{avail}}(t),\alpha_{\mathrm{crypto}}(t),\alpha_{\mathrm{deni}}(t)\big),
\]
with envelopes
\[
\alpha_{\Sigma}(t)\le \erisk(t)\le 1.
\]
Any violation forces fail-closed credit on that window.
\end{definition}

\begin{definition}[Risk-penalty channel map]\DefLabel{def:risk_penalty_channel_map}\label{def:risk_penalty_channel_map}
Each channel debit induces a nonnegative penalty contribution via pinned nondecreasing maps
\[
\psi_{\mathrm{ref}},\ \psi_{\mathrm{avail}},\ \psi_{\mathrm{crypto}},\ \psi_{\mathrm{deni}}:[0,1]\to\R_{\ge 0},\qquad \psi_j(0)=0.
\]
Define
\[
\PiRisk_t:=\psi_{\mathrm{ref}}(\alpha_{\mathrm{ref}}(t))+\psi_{\mathrm{avail}}(\alpha_{\mathrm{avail}}(t))+\psi_{\mathrm{crypto}}(\alpha_{\mathrm{crypto}}(t))+\psi_{\mathrm{deni}}(\alpha_{\mathrm{deni}}(t)).
\]
\end{definition}

\begin{proposition}[Risk-penalty monotonicity]\DefLabel{prop:risk_penalty_monotonicity}\label{prop:risk_penalty_monotonicity}
Under Definition~\ref{def:risk_penalty_channel_map}, increasing any channel debit while holding others fixed cannot decrease $\PiRisk_t$.
\end{proposition}

\begin{theorem}[Dual risk-ledger consistency]\DefLabel{thm:dual_risk_ledger_consistency}\label{thm:dual_risk_ledger_consistency}
If each window satisfies Definition~\ref{def:dual_risk_ledger} and horizon policy enforces $\sum_t \erisk(t)\le \bar\epsilon$, then
\[
\sum_t \alpha_{\Sigma}(t)\le \sum_t \erisk(t)\le \bar\epsilon.
\]
Hence channel-wise false-acceptance accounting is conservatively dominated by the reflection risk ledger.
\end{theorem}

\begin{proposition}[Per-window probability coherence]\DefLabel{prop:dual_risk_probability_coherence}\label{prop:dual_risk_probability_coherence}
If any window reports $\alpha_{\Sigma}(t)>1$ or $\erisk(t)\notin[0,1]$, the window is probability-incoherent and all credit-bearing claims on that window \MUST{} fail closed.
\end{proposition}

\begin{definition}[Conservative uncertainty set]\DefLabel{def:conservative_uncertainty_set}\label{def:conservative_uncertainty_set}
A family $\widehat{\Et}$ is conservative if $\Et \subseteq \widehat{\Et}$ for all $t$ and $\widehat{\Et}$ is computable from strict artifacts under pinned verifiers.
\end{definition}

\begin{definition}[Auditable size functional]\DefLabel{def:size_functional}\label{def:size_functional}
A size functional $S(\cdot)$ maps conservative uncertainty sets to $\R_{\ge 0}$ such that:
(i) $S$ is computable from strict artifacts, (ii) $A\subseteq B \Rightarrow S(A)\le S(B)$, and
(iii) $S$ upper bounds a relevant worst-case looseness term in certified inequalities.
\end{definition}

\begin{definition}[Localized regularization]\DefLabel{def:localized_regularization}\label{def:localized_regularization}
A localized regularization partitions a conservative set into a non-singular core $\widehat{\Et}^{\mathrm{core}}$ and a singular boundary escrow $\partial\widehat{\Et}^{\mathrm{sing}}$, with pinned rules specifying which inferences may be credited on the core and how escrow mass is tracked conservatively.
\end{definition}

\begin{definition}[Computable localization]\DefLabel{def:computable_localization}\label{def:computable_localization}
Computable localization is a pinned, budgeted procedure that constructs $\widehat{\Et}^{\mathrm{core}}$ and $\partial\widehat{\Et}^{\mathrm{sing}}$ without exact singular-boundary computation.
It \MUST{} (i) use replay-verifiable sampling/sketches, (ii) treat uncertain regions as escrow (never as core), and (iii) certify explicit error bars and compute cost.
\end{definition}

\begin{assumption}[Regularity certificate]\DefLabel{ass:regularity_certificate}\label{ass:regularity_certificate}
Each regime pins a regularity certificate that provides conservative local identifiability bounds on the computed core, flags degeneracy regions, and certifies that the localization procedure is computable within the regime's metrology budget.
If the certificate cannot be verified, the default behavior is fail-closed for deductive regularity claims; however, Section~\ref{sec:reflection} defines an optional empirical reflection gate that permits limited progress under metered soundness risk.
\end{assumption}

\begin{definition}[Contraction response certificate]\DefLabel{def:contraction_response_certificate}\label{def:contraction_response_certificate}
A contraction response certificate publishes replay-verifiable coefficients $(\kappa_t,\eta_t)$ with $\kappa_t>0,\eta_t\ge 0$ such that
\[
S\!\left(\widehat{\mathcal{E}_{t+1}}^{\mathrm{core}}\right)
\le
S\!\left(\widehat{\mathcal{E}_{t}}^{\mathrm{core}}\right)-\kappa_t\,\mathrm{ecr\_net}_t+\eta_t.
\]
The term $\eta_t$ is an explicit residual slack caused by finite-sample/modeling uncertainty.

\begin{lemma}[From ECR to contraction]\DefLabel{lem:ecr_to_contraction}\label{lem:ecr_to_contraction}
Fix a window $t=t_0,\dots,t_1-1$ in one observability regime.
Assume all mandatory gates pass on each $t$ (Theorem~\ref{thm:gate_vector_completeness}).
Assume the replay kernel yields $\mathbf{g}_t=1$ and $\mathrm{ecr\_net}_t\ge \underline{n}>0$.

\[
S\!\left(\widehat{\mathcal{E}_{t+1}}^{\mathrm{core}}\right)\le S\!\left(\widehat{\mathcal{E}_{t}}^{\mathrm{core}}\right)-\kappa_t\,\mathrm{ecr\_net}_t+\eta_t.
\]
Then
\[
S\!\left(\widehat{\mathcal{E}_{t_1}}^{\mathrm{core}}\right)\le S\!\left(\widehat{\mathcal{E}_{t_0}}^{\mathrm{core}}\right)-\underline{n}\sum_{t=t_0}^{t_1-1}\kappa_t+\sum_{t=t_0}^{t_1-1}\eta_t.
\]
\end{lemma}

\begin{theorem}[Identifiability takeoff]\DefLabel{thm:identifiability_takeoff}\label{thm:identifiability_takeoff}
Assume \lbl{ass:ledger_observable_only} and a fixed observability regime on a window.

\[
\underline{n}\sum_{t=t_0}^{t_1-1}\kappa_t>\sum_{t=t_0}^{t_1-1}\eta_t,
\]
then:
(i) $S(\widehat{\mathcal{E}_{t}}^{\mathrm{core}})$ contracts by at least $\underline{n}\sum_{t=t_0}^{t_1-1}\kappa_t-\sum_{t=t_0}^{t_1-1}\eta_t$, and
(ii) every \STE{}-certified lower bound that is monotone nonincreasing in conservative-set looseness is weakly improved on that window.
\end{theorem}
